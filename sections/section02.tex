%----------------------------------------------------------------------------------------
\begin{frame}[fragile]
    \frametitle{Python}
    
    \begin{python}
def calcular_doble(x):
    """Retorna el doble del número"""
    return 2 * x

# test de función
numero = 5
resultado = calcular_doble(numero)
print(f"El doble de {numero} es {resultado}")
    \end{python}
\end{frame}
